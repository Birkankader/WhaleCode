export const commands = {

  task_id: string;
  worktree_name: string;
  branch_name: string;

  created_at: string;
}

  path: string;
}

export interface ConflictReport {
  has_conflicts: boolean;
  conflicting_files: ConflictFile[];
  worktree_a: string;
  worktree_b: string;
}
