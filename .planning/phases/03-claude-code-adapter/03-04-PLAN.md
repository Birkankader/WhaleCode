**useClaudeTask.ts — Retry loop with spawnOnce pattern:**

Extract the current spawn logic into a `spawnOnce` helper that returns a Promise. Wrap the Channel-based spawn in a Promise constructor — resolve when the `exit` event fires:

```typescript
const spawnOnce = async (prompt: string, projectDir: string): Promise<{
  taskId: string | null;
  hitRateLimit: boolean;
  silentFailure: boolean;
}> => {
  return new Promise(async (resolve) => {
    let hitRateLimit = false;
    let lastResultJson: string | null = null;
    let taskId: string | null = null;

    const channel = new Channel<OutputEvent>();
    channel.onmessage = async (event) => {
      if ('Stdout' in event) {
        // Try to capture result events for validation
        try {
          const parsed = JSON.parse(event.Stdout);
          if (parsed.type === 'result') {
            lastResultJson = event.Stdout;
          }
        } catch { /* not JSON, ignore */ }
        // ... existing stdout handling ...
      } else if ('Stderr' in event) {
        // ... existing stderr handling, including rate limit detection ...
        if (/* rate limit detected */) {
          hitRateLimit = true;
        }
      } else if ('Exit' in event) {
        let isSilentFailure = false;
        // Validate result on exit
        if (lastResultJson) {
          try {
            await commands.validateClaudeResult(lastResultJson);
          } catch {
            isSilentFailure = true;
          }
        } else if (event.Exit === 0) {
          // Exit 0 but no result event = silent failure
          isSilentFailure = true;
        }
        resolve({ taskId, hitRateLimit, silentFailure: isSilentFailure });
      }
    };

    taskId = await commands.spawnClaudeTask(prompt, projectDir, channel);
  });
};
```

The outer `spawnTask` calls `spawnOnce` in a retry loop:

```typescript
const spawnTask = useCallback(async (prompt: string, projectDir: string) => {
  const policy = { maxRetries: 3, baseDelay: 5000, maxDelay: 60000 };

  for (let attempt = 0; attempt <= policy.maxRetries; attempt++) {
    if (attempt > 0) {
      const delay = Math.min(policy.baseDelay * 2 ** (attempt - 1), policy.maxDelay);
      setRateLimitWarning(`Retrying in ${delay / 1000}s (attempt ${attempt}/${policy.maxRetries})...`);
      await new Promise(r => setTimeout(r, delay));
    }

    setRateLimitWarning(false);
    const result = await spawnOnce(prompt, projectDir);

    if (result.silentFailure) {
      setSilentFailure(true);
    }

    if (!result.hitRateLimit) {
      return result.taskId;
    }

    // Rate limit hit — cancel and retry
    if (result.taskId) {
      await commands.cancelProcess(result.taskId);
    }
  }

  setRateLimitWarning('Rate limit retry exhausted after 3 attempts');
  return null;
}, []);
```

Change `rateLimitWarning` state type from `boolean` to `string | false`:
